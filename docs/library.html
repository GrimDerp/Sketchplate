var fs = require('fs'),
  Fetch = require('./fetch').Fetch;


module.exports = function buildLibrary( settings, key ){
  var libs = [];
  for( var id in settings[key]){
    var libInfo = {};
    for( var prop in settings[key][id] ){
      libInfo[prop] = settings[key][id][prop];
    }
    libInfo.id = id;
    libInfo.target = __dirname + '/../'+ settings.templateDirectory +'/'+ settings.javascriptsDirectory +'/'+ libInfo.target;
    libs.push( new Library( libInfo ) );
  }
  return libs;
};




var Library = function ( params ){
  for( var prop in params ){
    this[prop] = params[prop];
  }
};

Library.prototype = {
  constructor: Library,
  exists: function (){
    try {
      fs.statSync( this.target );
    } catch ( e ){
      return false;
    }
    return true;
  },
  fetch: function ( fn ){
    var ftch = new Fetch();
    if( fn )ftch.on('exit', fn );
    return ftch.retrieveLib( this );
  }
};
